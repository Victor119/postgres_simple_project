Table employees {
  employee_id int [pk, increment]
  first_name varchar
  last_name varchar
  cnp varchar
  username varchar
  password varchar
  email varchar
  address varchar
  city varchar
  country varchar
}

Table salaries {
  salary_id int [pk, increment]
  employee_id int [ref: > employees.employee_id]
  base_salary decimal
  month date
}

Table bonuses {
  bonus_id int [pk, increment]
  employee_id int [ref: > employees.employee_id]
  amount decimal
  description varchar
  month date
}

Table vacations {
  vacation_id int [pk, increment]
  employee_id int [ref: > employees.employee_id]
  start_date date
  end_date date
  number_of_days int
  reason varchar
}

Table work_days {
  work_day_id int [pk, increment]
  employee_id int [ref: > employees.employee_id]
  month date
  number_of_days int
}

Table archived_files {
  file_id int [pk, increment]
  employee_id int [ref: > employees.employee_id]
  file_name varchar
  file_type varchar  // 'pdf' or 'excel'
  path varchar
  sent_date date
}
