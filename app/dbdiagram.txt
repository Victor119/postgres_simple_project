Table users {
  user_id int [pk, increment]
  username varchar [unique, not null]
  password varchar [not null] // Parolă criptată
  email varchar [unique, not null]
  role varchar [not null] // 'admin', 'manager', 'employee'
  created_at timestamp
}

Table employees {
  employee_id int [pk, increment]
  user_id int [ref: > users.user_id] // Legătură opțională cu users
  first_name varchar
  last_name varchar
  cnp varchar [unique, not null]
  username varchar
  password varchar
  email varchar
  address varchar
  city varchar
  country varchar
}

Table salaries {
  salary_id int [pk, increment]
  employee_id int [ref: > employees.employee_id]
  base_salary decimal
  month date
}

Table bonuses {
  bonus_id int [pk, increment]
  employee_id int [ref: > employees.employee_id]
  amount decimal
  description varchar
  month date
}

Table vacations {
  vacation_id int [pk, increment]
  employee_id int [ref: > employees.employee_id]
  start_date date
  end_date date
  number_of_days int
  reason varchar
}

Table work_days {
  work_day_id int [pk, increment]
  employee_id int [ref: > employees.employee_id]
  month date
  number_of_days int
}

Table archived_files {
  file_id int [pk, increment]
  employee_id int [ref: > employees.employee_id]
  file_name varchar
  file_type varchar // 'pdf' or 'excel'
  path varchar
  sent_date date
}
